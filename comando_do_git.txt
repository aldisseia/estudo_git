git branch
	mostra o ramo atual que o projeto encontra-se
git checkout -b novo_ramo
	cria um novo ramo no grafo
git log --oneline
	mostra o projeto com seus dados
git checkout id
	volta para a verção anterios do ID
git checkout "nome do branch"
	retorn para o estado atual ou mais recente do arquivo
git diff
	mostra as mudanças que foram feitas no projeto
"Em vermelho é o que foi removido, em verde o que foi acresetado"
git resert --hard id 
	volta para verção do id
